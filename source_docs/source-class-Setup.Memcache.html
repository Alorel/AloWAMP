 14       /**

 17        * @var string

 21       /**
 22        * Constructor
 23        *
 24        * @author Art <a.molcanovas@gmail.com>
 25        */
 26       function __construct() {
 27          $this->dest = DIR_TMP . 'memcache.zip';
 28          $this->dest_unzip = DIR_TMP . 'AloWAMP-bin-memcached' . DIRECTORY_SEPARATOR;
 29          $this->download()
 30             ->unzip()
 31             ->copy()
 32             ->installService();
 33       }

 35       /**
 36        * Installs the service
 37        *
 38        * @author Art <a.molcanovas@gmail.com>
 39        * @return Memcache
 40        */
 41       protected function installService() {
 42          if (Service::exists('alomemcache')) {
 43             _echo('Removing previous AloWAMP Memcache service');
 44             Service::delete('alomemcache');
 45          }
 46 
 47          _echo('Installing Memcache service');
 48          Service::installExe('alomemcache', DIR_MEMCACHE . 'memcached.exe -d runservice', 'AloWAMP Memcache');
 49 
 50          return $this;
 51       }
 52 
 53       /**
 54        * Copies unzipped contents
 55        *
 56        * @author Art <a.molcanovas@gmail.com>
 57        * @return Memcache
 58        */
 59       protected function copy() {
 60          _echo('Copying unzipped contents..');
 61 
 62          shell_exec('xcopy /s /e "' . rtrim($this->dest_unzip, DIRECTORY_SEPARATOR) . '" "' . rtrim(DIR_INDEX, DIRECTORY_SEPARATOR) . '"');
 63 
 64          if (file_exists(DIR_INDEX . 'README.md')) {
 65             unlink(DIR_INDEX . 'README.md');
 66          }
 67 
 68          if (file_exists(DIR_MEMCACHE)) {
 69             _echo('Copy successful!');
 70             $this->cleanup();
 71          } else {
 72             die('Failed to copy. Terminating setup.');
 73          }
 74 
 75          return $this;
 76       }
 77 
 78       /**
 79        * Unzips contents
 80        *
 81        * @author Art <a.molcanovas@gmail.com>
 82        * @return Memcache
 83        */
 84       protected function unzip() {
 85          _echo('Unzipping...');
 86          $zip = new \ZipArchive();
 87          $res = $zip->open($this->dest);
 88 
 89          if ($res === true) {
 90             if (file_exists($this->dest_unzip)) {
 91                rmdir($this->dest_unzip);
 92             }
 93 
 94             $zip->extractTo(DIR_TMP);
 95             $zip->close();
 96          } else {
 97             die('Failed to unzip ' . $this->dest . '. Terminating.');
 98          }
 99 
100          return $this;
101       }
102 
103       /**

109       protected function download() {
110          $this->downloader = new \Downloader(self::MEMCACHE_DOWNLOAD_SOURCE, $this->dest);
111 
112          if ($this->downloader->download()) {
113             _echo('Download successful. Setting up Memcache...');
114          } else {
115             die('Download failed. Aborting setup.');
116          }
117 
118          return $this;
119       }
