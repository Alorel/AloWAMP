 70        * Updates dependencies

 72        * @author Art <a.molcanovas@gmail.com>
 73        * @return AbstractVersionSwitch
 74        */
 75       function updateDependencies() {
 76          $allowed = [DIR_APACHE];
 77 
 78          _echo('Checking dependencies');
 79 
 80          foreach ($this->dependent_dirs as $d) {
 81             if (file_exists($d) && in_array($d, $allowed)) {
 82                $scan = scandir($d);
 83                Format::formatScandir($scan);
 84 
 85                if (!empty($scan)) {
 86                   $method = $service = false;
 87 
 88                   switch ($d) {
 89                      case DIR_APACHE:
 90                         $method = 'updateApache';
 91                         $service = 'aloapache';
 92                         break;
 93                   }
 94 
 95                   if ($method) {
 96                      foreach ($scan as $version) {
 97                         call_user_func([$this, $method], $version);
 98                      }
 99 
100                      if ($service && Service::exists($service)) {
101                         _echo('Restarting service');
102                         Service::restart($service);
103                      }
104                   }
105                }
106             }
107          }
108 
109          _echo('Switch complete');
110

134       protected function updateApache($version) {
135          _echo('Updating Apache ' . $version);
136          $file = DIR_APACHE . $version . DIRECTORY_SEPARATOR . 'conf' . DIRECTORY_SEPARATOR . 'httpd.conf';
137 
138          if (!file_exists($file)) {
139             _echo('Failed to update ' . $version . ': httpd.conf not found');
140          } else {
141             $contents = file_get_contents($file);
142             $contents = str_ireplace('bin' . DIRECTORY_SEPARATOR . 'php' . DIRECTORY_SEPARATOR . $this->old_version, 'bin' . DIRECTORY_SEPARATOR . 'php' . DIRECTORY_SEPARATOR . $this->new_version, $contents);
143 
144             if (file_put_contents($file, $contents) !== false) {
145                _echo($version . ' updated');
146             } else {
147                _echo('Failed to edit ' . $version . '\'s httpd.conf');
148             }
149          }

152       /**
153        * Attempts to switch versions

156        * @return AbstractVersionSwitch

158       function trySwitch() {
159          if (!file_exists($this->dir)) {
160             die('The module is not installed');
161          } elseif (!$this->old_version) {
162             die('Module installation config entry missing');
163          } else {
164             $installed_versions = scandir($this->dir);
165             Format::formatScandir($installed_versions);
166 
167             if (count($installed_versions) < 2) {
168                die('You only have one module version installed');
169             } else {
170                _echo('Your current version is ' . $this->old_version . '. Which version would you like to switch to? Available options are:');
171 
172                foreach ($installed_versions as $v) {
173                   echo "\t$v" . PHP_EOL;
174                }
175 
176                $io = IO::readline();
177 
178                if (!$io) {
179                   return $this->trySwitch();
180                } elseif (!in_array($io, $installed_versions)) {
181                   _echo('You don\'t have that version installed');
182 
183                   return $this->trySwitch();
184                } else {
185                   SET::$s->{$this->ini} = $this->new_version = $io;
186                   SET::$s->save();
187                   $this->updateDependencies()->restartService();
188                }
189             }
190          }
191 
192          return $this;
193       }
