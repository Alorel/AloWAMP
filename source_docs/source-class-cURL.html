204       function wasSuccessful() {
205          if ($this->errno !== CURLE_OK) {
206             return 'cURL error #' . $this->errno;
207          } else {
208             $code = $this->getinfo(CURLINFO_HTTP_CODE);
209             if (!in_array(substr('' . $this->getinfo(CURLINFO_HTTP_CODE), 0, 1), [1, 2, 3])) {
210                return 'HTML response ' . $code;
211             } else {
212                return true;
213             }
214          }
